    // Goroutine wrapper
    class GoroutineObj : public RuntimeObject {
    public:
        Closure* closure;          // Function to run in goroutine
        std::vector<Value> args;   // Arguments to pass
        std::atomic<bool> done;    // Completion flag
        Value result;              // Return value (if any)
        
        GoroutineObj(Closure* c, std::vector<Value> a)
            : RuntimeObject(ObjectType::GOROUTINE), closure(c), 
              args(std::move(a)), done(false), result(Value::undefined()) {}
        
        std::string toString() const override { return "[Goroutine]"; }
    };
    
    // Channel — mutex-protected buffered queue
    class ChannelObj : public RuntimeObject {
    public:
        std::vector<Value> buffer;
        int capacity;
        std::mutex mtx;
        std::condition_variable sendCV;
        std::condition_variable recvCV;
        bool closed;
        
        ChannelObj(int cap = 0) : RuntimeObject(ObjectType::CHANNEL), 
            capacity(cap), closed(false) {}
        
        void close();
        bool isFull();
        bool isEmpty();
        std::string toString() const override { return "[Channel]"; }
    };
    
    // Mutex — wrapper around pthread_mutex_t
    class MutexObj : public RuntimeObject {
    public:
        pthread_mutex_t mutex;
        
        MutexObj() : RuntimeObject(ObjectType::MUTEX) {
            pthread_mutex_init(&mutex, nullptr);
        }
        
        ~MutexObj() {
            pthread_mutex_destroy(&mutex);
        }
        
        void lock() { pthread_mutex_lock(&mutex); }
        void unlock() { pthread_mutex_unlock(&mutex); }
        
        std::string toString() const override { return "[Mutex]"; }
    };
    
    // WaitGroup — atomic counter with condition variable
    class WaitGroupObj : public RuntimeObject {
    public:
        std::atomic<int> counter;
        std::mutex mtx;
        std::condition_variable cv;
        
        WaitGroupObj() : RuntimeObject(ObjectType::WAITGROUP), counter(0) {}
        
        void add(int delta);
        void wait();
        void done();
        std::string toString() const override { return "[WaitGroup]"; }
    };

    class GoroutineScheduler {
    public:
        std::queue<std::function<void()>> workQueue;
        std::mutex queueMtx;
        std::condition_variable queueCV;
        std::atomic<bool> shutdown;
        int threadCount;
        
        GoroutineScheduler(int count) : threadCount(count), shutdown(false) {}
        
        void start(std::function<void()> workerFn);
        void submit(std::function<void()> work);
        void stop();
    };
    
    Implement VM methods:
    
    initScheduler(threadCount):
    - Create new GoroutineScheduler
    - Set running = true
    - Spawn threadCount worker threads
    - Workers call workerLoop()
    
    workerLoop():
    - While running and !shutdown:
      - Wait on queueCV for work
      - If work available: pop and execute
      - If no work: wait with timeout
    
    shutdownScheduler():
    - Set shutdown = true, notify all
    - Join all worker threads
    - Delete scheduler
    
    spawnGoroutine(closure, args):
    - Create GoroutineObj(closure, args)
    - Submit lambda to scheduler that:
      - Calls closure with args
      - Sets done = true on GoroutineObj
      - Stores return value
    
    channelSend(ch, value):
    - Lock ch->mtx
    - If ch->closed: panic("send on closed channel")
    - If ch->buffer.size() < ch->capacity: push value, notify recvCV
    - Else: wait on sendCV (release lock while waiting)
    - Unlock ch->mtx
    
    channelReceive(ch):
    - Lock ch->mtx
    - If ch->buffer empty:
      - If ch->closed: pop default/undefined, notify sendCV
      - Else: wait on recvCV
    - Pop from buffer, notify sendCV
    - Unlock ch->mtx
    
    channelClose(ch):
    - Lock ch->mtx
    - Set ch->closed = true
    - Notify all sendCV and recvCV
    - Unlock ch->mtx
    
    Also implement mutexLock, mutexUnlock, waitGroupAdd, waitGroupWait, selectExecute.
