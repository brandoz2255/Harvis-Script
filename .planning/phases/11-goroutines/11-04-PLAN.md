  <name>Task 1: Add compiler visitor methods for concurrency AST nodes</name>
  <files>src/compiler/compiler.h, src/compiler/compiler.cpp</files>
  <action>
    In compiler.h, ensure Compiler class inherits from Visitor and StatementVisitor (it should already).
    
    In compiler.cpp, implement:
    
    visitGoStmt(GoStmt* stmt):
    - stmt->function->accept(*this)  // Push function/closure on stack
    - For each arg in stmt->args: arg->accept(*this)  // Push args on stack
    - emitBytes(Opcode::OP_GO, static_cast<uint8_t>(stmt->args.size()))
    
    visitChannelExpr(ChannelExpr* expr):
    - If expr->capacity is not null: capacity->accept(*this)  // Push capacity
    - Else: emitByte(Opcode::OP_CONST_NUMBER), emitConstant(Value(0.0))  // Default unbuffered
    - emitByte(Opcode::OP_MAKE_CHANNEL)
    
    visitSendExpr(SendExpr* expr):
    - expr->channel->accept(*this)  // Push channel
    - expr->value->accept(*this)  // Push value
    - emitByte(Opcode::OP_SEND)
    
    visitReceiveExpr(ReceiveExpr* expr):
    - expr->channel->accept(*this)  // Push channel
    - emitByte(Opcode::OP_RECEIVE)
    - If commaOk, emit additional OP pair to handle ok boolean
    
    visitSelectStmt(SelectStmt* stmt):
    - For each case in stmt->cases:
      - If isSend: emit channel value, then value, tag as send
      - If isReceive: emit channel, tag as receive
      - If isDefault: emit tag as default
    - After all branches: emit OP_SELECT
    - For each case body: emit case body statements (they execute inline)
    
    visitMutexExpr(MutexExpr* expr):
    - emitByte(Opcode::OP_MUTEX_NEW)
    
    visitWaitGroupExpr(WaitGroupExpr* expr):
    - emitByte(Opcode::OP_WAITGROUP_NEW)
  </action>

<task type="auto">
  <name>Task 2: Add VM opcode dispatch for concurrency opcodes</name>
  <files>src/vm/vm.cpp</files>
  <action>
    In VM::interpret(), add cases to the main opcode switch:
    
    case Opcode::OP_GO: {
        // Pop arg count, pop closure, spawn goroutine
        uint8_t argCount = readByte();
        Closure* closure = (Closure*)stackPop().asPointer();
        std::vector<Value> args(argCount);
        for (int i = argCount - 1; i >= 0; i--) args[i] = pop();
        spawnGoroutine(closure, args);
        break;
    }
    
    case Opcode::OP_MAKE_CHANNEL: {
        // Pop capacity, create channel
        double cap = pop().asNumber();
        ChannelObj* ch = new ChannelObj(static_cast<int>(cap));
        push(Value::of(ch));
        activeChannels.push_back(ch);
        break;
    }
    
    case Opcode::OP_SEND: {
        // Pop value, pop channel, send (blocking)
        Value value = pop();
        ChannelObj* ch = (ChannelObj*)pop().asPointer();
        channelSend(ch, value);
        break;
    }
    
    case Opcode::OP_RECEIVE: {
        // Pop channel, receive (blocking), push value
        ChannelObj* ch = (ChannelObj*)pop().asPointer();
        Value result = channelReceive(ch);
        push(result);
        break;
    }
    
    case Opcode::OP_SELECT: {
        // Pop branch count, execute select
        uint8_t branchCount = readByte();
        // Read branch metadata from chunk
        // Execute selectExecute with collected branches
        int chosen = selectExecute(sends, receives);
        // Jump to chosen branch or default
        break;
    }
    
    case Opcode::OP_MUTEX_NEW: {
        // Push new mutex
        MutexObj* m = new MutexObj();
        push(Value::of(m));
        break;
    }
    
    case Opcode::OP_MUTEX_LOCK: {
        // Pop mutex, lock
        MutexObj* m = (MutexObj*)pop().asPointer();
        mutexLock(m);
        break;
    }
    
    case Opcode::OP_MUTEX_UNLOCK: {
        // Pop mutex, unlock
        MutexObj* m = (MutexObj*)pop().asPointer();
        mutexUnlock(m);
        break;
    }
    
    case Opcode::OP_WAITGROUP_NEW: {
        // Push new waitgroup
        WaitGroupObj* wg = new WaitGroupObj();
        push(Value::of(wg));
        break;
    }
    
    case Opcode::OP_WAITGROUP_ADD: {
        // Pop delta, pop waitgroup, add delta
        double delta = pop().asNumber();
        WaitGroupObj* wg = (WaitGroupObj*)pop().asPointer();
        waitGroupAdd(wg, static_cast<int>(delta));
        break;
    }
    
    case Opcode::OP_WAITGROUP_WAIT: {
        // Pop waitgroup, block until zero
        WaitGroupObj* wg = (WaitGroupObj*)pop().asPointer();
        waitGroupWait(wg);
        break;
    }
  </action>
