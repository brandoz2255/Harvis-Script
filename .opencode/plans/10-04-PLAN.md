            auto* mapObj = reinterpret_cast<MapObj*>(obj.asObject());

  <name>Task 3: Implement OP_DEFER, OP_DEFERRED_RETURN, OP_PANIC, OP_RECOVER</name>
  <files>src/vm/vm.cpp</files>
  <action>
    Implement in VM::interpret() switch:
    
    case Opcode::OP_DEFER: {
        // Operand: none
        Value callable = pop();  // callable on top of stack
        deferredCallStack.push_back({callable, {}});
        break;
    }
    
    case Opcode::OP_DEFERRED_RETURN: {
        // Execute deferred calls in LIFO order
        for (int i = static_cast<int>(deferredCallStack.size()) - 1; i >= 0; i--) {
            auto& dc = deferredCallStack[i];
            // Call the deferred callable (it's a closure or native function)
            push(dc.callable);
            // Simple call with 0 args for now
            Value callee = peek(0);
            if (callee.isObject() && callee.asObject()) {
                auto* obj = callee.asObject();
                if (obj->type == ObjectType::Closure) {
                    auto* closure = reinterpret_cast<Closure*>(obj);
                    CallFrame frame;
                    frame.function = closure->function;
                    frame.chunk = &closure->function->chunk;
                    frame.ip = 0;
                    frame.stackStart = static_cast<int>(stack.size()) - 1;
                    callStack.push(frame);
                    interpret();  // Recursive call to run deferred function
                    callStack.pop();
                }
            }
        }
        deferredCallStack.clear();
        break;
    }
    
    case Opcode::OP_PANIC: {
        // Operand: none
        panicValue = pop();
        panicActive = true;
        reportError("Panic: " + panicValue.toString());
        return false;  // Signal error to top-level interpret
    }
    
    case Opcode::OP_RECOVER: {
        // Operand: none
        recoverValue = panicActive ? panicValue : Value();
        panicActive = false;
        panicValue = Value();
        push(recoverValue);
        break;
    }
    
    Modify VM::interpret() to check panicActive at the top of the loop:

        // Unwind to nearest recover point
        // For now, just clear panic and continue
        panicActive = false;
        continue;
    }
